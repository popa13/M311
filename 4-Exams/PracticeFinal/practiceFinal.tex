\question 

Let $T (x, y, z) = (x , -y, x)$.
  \begin{parts}
  \part[5]
  Show that $T$ is a linear transformation.
  \part[5]
  Find the kernel and the image of $T$.
  \part[5]
  Find the nullity and the rank of $T$.
  \end{parts}

\question 

Let $U = \mathrm{span} \, \{ (1, 1, 1) , (-1, 0, 2) \}$.
\begin{parts}
\part[5]
Are $(1, 1, 1)$ and $(-1, 0, 2)$ orthogonal?
\part[5]
Is the vector $(2, -3, 1)$ in $U$?
\part[5]
Using the Gram-Schmidt process, transform the set $\{ (1, 1, 1) , (-1, 0, 2) \}$ into a set $F$ of orthogonal vectors.
\end{parts}


\newpage 

\question 

Let $V = \mathbb{R}^3$. Let $B$ be the standard basis for $\mathbb{R}^3$ and let $D = \{ (1, 1, 0) , (1, 0, 1) , (0, 1, 0) \}$ be another basis of $\mathbb{R}^3$. Let $T (a, b, c) = (2a - b, b + c, c - 3a)$ be a linear operator.

  \begin{parts}
  \part[5]
  Find the change matrix $P_{B \leftarrow D}$.
  \part[5]
  Find the matrix representation of $T$ in the basis $B$.
  \part[5]
  Knowing that
    \[
      P_{D \leftarrow B} = \begin{bmatrix} 1 & 0 & -1 \\ 0 & 0 & 1 \\ -1 & 1 & 1 \end{bmatrix} ,
    \]
  find the matrix representation of the linear operator $T$ in the basis $D$. \\ 
  \textit{[Hint: $P^{-1}_{D \leftarrow B} M_D (T) P_{D \leftarrow B} = M_B (T)$ and $P^{-1}_{D \leftarrow B} = P_{B \leftarrow D}$.]}
  \end{parts}

\newpage 

\phantom{2}

\question

Let $T : \mathbf{P_n} \ra \mathbf{P_n}$ be the \textbf{linear} operator $T (p (x)) = p (x) - p (-x)$.
  \begin{parts}
  \part[3]
  Show that $\ker T = \{ p \, : \, p (x) = p (-x) \}$ and $\mathrm{im} \, T = \{ q \, : \, q (-x) = -q (x) \}$.
  \part[2]
  If $U$ is the subspace of all \textbf{even}\footnote{Recall that a function $f(x)$ is \textit{even} if $f (-x) = f (x)$.} polynomials and $V$ is the subspace of all \textbf{odd}\footnote{Recall that a function $f(x)$ is \textit{odd} if $f (-x) = -f (x)$.} polynomials, then show that
    \[
      n + 1 = \dim U + \dim V .
    \]
  \end{parts}
