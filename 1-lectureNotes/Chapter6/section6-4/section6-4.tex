\section{Finite Dimension}

\begin{example}
\begin{enumerate}[label=\alph*)]
	\item The vector space $\mathbb{R}^m$ is a finite dimensional space because it is spanned by a finite set of vectors, that is $\{ \mathbf{e_1} , \mathbf{e_2} , \ldots , \mathbf{e_m} \}$.
	\item The vector space $\mathbf{P}_n$ is a finite dimensional space because it is spanned by a finite set of vectors, that is $\{ 1 , x , x^2 , \ldots , x^n \}$.
\end{enumerate}
\end{example}

\begin{definition}
A vector space $V$ is called \textbf{finite dimensional} if it is spanned by a finite set of vectors. Otherwise, it is called \textbf{infinite dimensional}.
\end{definition}
